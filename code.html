The Code for FIZZ BUZZ.


                        function getValues(){
                            
                            let fizzValue = document.getElementById("fizzValue").value;
                            let buzzValue = document.getElementById("buzzValue").value;
                    
                            fizzValue = parseInt(fizzValue);
                            buzzValue = parseInt(buzzValue);
                        
                            if(Number.isInteger(fizzValue) && Number.isInteger(buzzValue))    {
                                let fbData = FizzBuzz(fizzValue, buzzValue);

                                displayData(fbData);
                            } else {
                                alert("You must enter integers");
                            }
                        }
                        
                        function FizzBuzz(value1, value2) {
                            
                            let returnArray = [];
                        
                            for(let index = 1; index <= 100; index++){
                                
                                if(index % value1 == 0 && index % value2 == 0){
                                    returnArray.push("FizzBuzz");  
                                } 
                                else if(index % value1 == 0){
                                    returnArray.push("Fizz");   
                                } 
                                else if(index % value2 == 0){
                                    returnArray.push("Buzz");
                                } 
                                else
                                {
                                    returnArray.push(index);
                                } 
                            }
                            
                            return returnArray; 
                        }
                        
                        function displayData(fbData) {

                            let tableBody = document.getElementById("results");
                            let templateRow = document.getElementById("fbTemplate");

                            tableBody.innerHTML = "";
                        
                            for (let i = 0; i < fbData.length; i += 5) {
                                const tableRow = document.importNode(templateRow.content, true);

                                rowCols = tableRow.querySelectorAll("td");
                        
                                rowCols[0].classList.add(fbData[i]);
                                rowCols[0].textContent = fbData[i];
                               
                                rowCols[1].classList.add(fbData[i+ 1]);
                                rowCols[1].textContent = fbData[i + 1];
                               
                                rowCols[2].classList.add(fbData[i+2]);
                                rowCols[2].textContent = fbData[i + 2];
                               
                                rowCols[3].classList.add(fbData[i+3]);
                                rowCols[3].textContent = fbData[i + 3];
                               
                                rowCols[4].classList.add(fbData[i+4]);
                                rowCols[4].textContent = fbData[i + 4];
                        
                                tableBody.appendChild(tableRow);
                            }
                        }
getValues()

In getValues function we get the fizz and buzz values from the app page. Next we validate that the fizz and buzz values are integers. Once we have validated that fizz and buzz are integers we pass those values to FizzBuzz function to complete the logic of if the values between 1 to 100 are either a multiple of 3 or 5 or both. Once the fizz buzz logic is completed we pass the array of results to displayData function.

FizzBuzz()

In the FizzBuzz function we use the modules operations to determine if the value being processed is either a multple of 3, which is assigned Fizz, a multiple of 5, which is assigned Buzz, or a multiple of both 3 and 5, which is assigned FizzBuzz. The results for each number between 1 to 100 is stored in an array and that array is passed back to getValues function.

displayData()

In displayData function we use the HTML template element to store the table format that we will copy and insert the array from getValues function. The template element is designed to have 5 columns per row so the data is presented a better format than one column down the web page.
